117      @Override
118      public Resource init() throws ResourceInstantiationException {
119        if(tagger == null) {
120          try {
121            tagger = new MaxentTagger(modelFile.toExternalForm());
122          } catch(Exception e) {
123            throw new ResourceInstantiationException(e);
124          }
125        }
126        return this;
127      }

129      @Override
130      public void reInit() throws ResourceInstantiationException {
131        tagger = null;
132        init();
133      }

135      @Override
136      public void execute() throws ExecutionException {
137        // check the parameters
138        if(document == null)
139          throw new ExecutionException("No document to process!");
140    
141        AnnotationSet inputAS = document.getAnnotations(inputASName);
142    
143        if(baseTokenAnnotationType == null ||
144          baseTokenAnnotationType.trim().length() == 0) { throw new ExecutionException(
145          "No base Token Annotation Type provided!"); }
146    
147        if(baseSentenceAnnotationType == null ||
148          baseSentenceAnnotationType.trim().length() == 0) { throw new ExecutionException(
149          "No base Sentence Annotation Type provided!"); }
150    
151        if(outputAnnotationType == null ||
152          outputAnnotationType.trim().length() == 0) { throw new ExecutionException(
153          "No AnnotationType provided to store the new feature!"); }
154    
155        AnnotationSet sentencesAS = inputAS.get(baseSentenceAnnotationType);
156        AnnotationSet tokensAS = inputAS.get(baseTokenAnnotationType);
157        if(sentencesAS != null && sentencesAS.size() > 0 && tokensAS != null &&
158          tokensAS.size() > 0) {
159          long startTime = System.currentTimeMillis();
160          fireStatusChanged("POS tagging " + document.getName());
161          fireProgressChanged(0);
162          // prepare the input for MaxentTagger
163          List<Word> sentenceForTagger = new ArrayList<Word>();
164    
165          // define a comparator for annotations by start offset
166          OffsetComparator offsetComparator = new OffsetComparator();
167    
168          // read all the tokens and all the sentences
169          List<Annotation> sentencesList = new ArrayList<Annotation>(sentencesAS);
170          Collections.sort(sentencesList, offsetComparator);
171          List<Annotation> tokensList = new ArrayList<Annotation>(tokensAS);
172          Collections.sort(tokensList, offsetComparator);
173    
174          Iterator<Annotation> sentencesIter = sentencesList.iterator();
175          ListIterator<Annotation> tokensIter = tokensList.listIterator();
176    
177          List<Annotation> tokensInCurrentSentence = new ArrayList<Annotation>();
178          Annotation currentToken = tokensIter.next();
179          int sentIndex = 0;
180          int sentCnt = sentencesAS.size();
181          while(sentencesIter.hasNext()) {
182            Annotation currentSentence = sentencesIter.next();
183            tokensInCurrentSentence.clear();
184            sentenceForTagger.clear();
185            while(currentToken != null &&
186              currentToken.getEndNode().getOffset()
187                .compareTo(currentSentence.getEndNode().getOffset()) <= 0) {
188              // If we're only POS tagging Tokens within baseSentenceAnnotationType,
189              // don't add the sentence if the Tokens aren't within the span of
190              // baseSentenceAnnotationType
191              if(posTagAllTokens || currentToken.withinSpanOf(currentSentence)) {
192                tokensInCurrentSentence.add(currentToken);
193    
194                if(useExistingTags && currentToken.getFeatures().containsKey(
195                      TOKEN_CATEGORY_FEATURE_NAME)) {
196                  sentenceForTagger.add(new TaggedWord(
197                        (String)currentToken.getFeatures()
198                          .get(TOKEN_STRING_FEATURE_NAME),
199                        (String)currentToken.getFeatures()
200                          .get(TOKEN_CATEGORY_FEATURE_NAME)));
201                } else {
202                  sentenceForTagger.add(new Word((String)currentToken.getFeatures()
203                    .get(TOKEN_STRING_FEATURE_NAME)));
204                }
205              }
206              currentToken = (tokensIter.hasNext() ? tokensIter.next() : null);
207            }
208    
209            // if the sentence doesn't contain any tokens (which is a bit weird but
210            // is possible) then don't try running the POS tagger as you will get an
211            // array index out of bounds exception
212            if(sentenceForTagger.isEmpty()) continue;
213    
214            // run the POS tagger
215            ArrayList<TaggedWord> taggerResults =
216              tagger.tagSentence(sentenceForTagger, useExistingTags);
217    
218            // add the results
219            // make sure no malfunction occurred
220            if(taggerResults.size() != tokensInCurrentSentence.size())
221              throw new ExecutionException(
222                "POS Tagger malfunction: the output size (" +
223                  taggerResults.size() + ") is different from the input size (" +
224                  tokensInCurrentSentence.size() + ")!");
225            Iterator<TaggedWord> resIter = taggerResults.iterator();
226            Iterator<Annotation> tokIter = tokensInCurrentSentence.iterator();
227            while(resIter.hasNext()) {
228              Annotation annot = tokIter.next();
229              addFeatures(annot, TOKEN_CATEGORY_FEATURE_NAME, ((String)resIter
230                .next().tag()));
231            }
232            fireProgressChanged(sentIndex++ * 100 / sentCnt);
233          }// while(sentencesIter.hasNext())
234    
235          if(currentToken != null && posTagAllTokens) {
236            // Tag remaining Tokens if we are not considering those only within
237            // baseSentenceAnnotationType
238    
239            // we have remaining tokens after the last sentence
240            tokensInCurrentSentence.clear();
241            sentenceForTagger.clear();
242            while(currentToken != null) {
243              tokensInCurrentSentence.add(currentToken);
244              if(useExistingTags && currentToken.getFeatures().containsKey(
245                    TOKEN_CATEGORY_FEATURE_NAME)) {
246                sentenceForTagger.add(new TaggedWord(
247                      (String)currentToken.getFeatures()
248                        .get(TOKEN_STRING_FEATURE_NAME),
249                      (String)currentToken.getFeatures()
250                        .get(TOKEN_CATEGORY_FEATURE_NAME)));
251              } else {
252                sentenceForTagger.add(new Word((String)currentToken.getFeatures()
253                  .get(TOKEN_STRING_FEATURE_NAME)));
254              }
255              currentToken = (tokensIter.hasNext() ? tokensIter.next() : null);
256            }
257    
258            // run the POS tagger on remaining tokens
259            List<TaggedWord> taggerResults =
260              tagger.tagSentence(sentenceForTagger, useExistingTags);
261    
262            // add the results and make sure no malfunction occurred
263            if(taggerResults.size() != tokensInCurrentSentence.size())
264              throw new ExecutionException(
265                "POS Tagger malfunction: the output size (" + taggerResults.size() +
266                  ") is different from the input size (" +
267                  tokensInCurrentSentence.size() + ")!");
268            Iterator<TaggedWord> resIter = taggerResults.iterator();
269            Iterator<Annotation> tokIter = tokensInCurrentSentence.iterator();
270            while(resIter.hasNext()) {
271              Annotation annot = tokIter.next();
272              addFeatures(annot, TOKEN_CATEGORY_FEATURE_NAME, ((String)resIter
273                .next().tag()));
274            }
275          }// if(currentToken != null)
276          fireProcessFinished();
277          fireStatusChanged(document.getName() +
278            " tagged in " +
279            NumberFormat.getInstance().format(
280              (double)(System.currentTimeMillis() - startTime) / 1000) +
281            " seconds!");
282        } else {
283          if(failOnMissingInputAnnotations) {
284            throw new ExecutionException(
285              "No sentences or tokens to process in document " +
286                document.getName() + "\n" + "Please run a sentence splitter " +
287                "and tokeniser first!");
288          } else {
289            Utils
290              .logOnce(
291                logger,
292                Level.INFO,
293                "POS tagger: no sentence or token annotations in input document - see debug log for details.");
294            logger.debug("No input annotations in document " + document.getName());
295          }
296        }
297    
298      }

300      protected void addFeatures(Annotation annot, String featureName,
301                                 String featureValue) throws GateRuntimeException {
302        String tempIASN = inputASName == null ? "" : inputASName;
303        String tempOASN = outputASName == null ? "" : outputASName;
304        if(outputAnnotationType.equals(baseTokenAnnotationType) &&
305          tempIASN.equals(tempOASN)) {
306          annot.getFeatures().put(featureName, featureValue);
307          return;
308        } else {
309          int start = annot.getStartNode().getOffset().intValue();
310          int end = annot.getEndNode().getOffset().intValue();
311    
312          // get the annotations of type outputAnnotationType
313          AnnotationSet outputAS = document.getAnnotations(outputASName);
314          AnnotationSet annotations = outputAS.get(outputAnnotationType);
315          if(annotations == null || annotations.size() == 0) {
316            // add new annotation
317            FeatureMap features = Factory.newFeatureMap();
318            features.put(featureName, featureValue);
319            try {
320              outputAS.add(new Long(start), new Long(end), outputAnnotationType,
321                features);
322            } catch(Exception e) {
323              throw new GateRuntimeException("Invalid Offsets");
324            }
325          } else {
326            // search for the annotation if there is one with the same start and end
327            // offsets
328            ArrayList<Annotation> tempList =
329              new ArrayList<Annotation>(annotations.get());
330            boolean found = false;
331            for(int i = 0; i < tempList.size(); i++) {
332              Annotation annotation = (Annotation)tempList.get(i);
333              if(annotation.getStartNode().getOffset().intValue() == start &&
334                annotation.getEndNode().getOffset().intValue() == end) {
335                // this is the one
336                annotation.getFeatures().put(featureName, featureValue);
337                found = true;
338                break;
339              }
340            }
341    
342            if(!found) {
343              // add new annotation
344              FeatureMap features = Factory.newFeatureMap();
345              features.put(featureName, featureValue);
346              try {
347                outputAS.add(new Long(start), new Long(end), outputAnnotationType,
348                  features);
349              } catch(Exception e) {
350                throw new GateRuntimeException("Invalid Offsets");
351              }
352            }
353          }
354        }
355      }

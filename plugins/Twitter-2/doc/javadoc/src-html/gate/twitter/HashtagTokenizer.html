052      // A comparator that sorts annotations by length; longest first
053      private static Comparator<Annotation> lengthComparator =
054        new Comparator<Annotation>() {
055          @Override
056          public int compare(Annotation a1, Annotation a2) {
057            long l1 = a1.getEndNode().getOffset() - a1.getStartNode().getOffset();
058            long l2 = a2.getEndNode().getOffset() - a2.getStartNode().getOffset();
059    
060            return (int)(l2 - l1);
061          }
062        };

118      @Override
119      public Resource init() throws ResourceInstantiationException {
120    
121        // load and configure the hidden gazetteer
122        FeatureMap hidden = Factory.newFeatureMap();
123        Gate.setHiddenAttribute(hidden, true);
124    
125        FeatureMap params = Factory.newFeatureMap();
126        params.put("listsURL", gazURL);
127        params.put("caseSensitive", Boolean.FALSE);
128    
129        params.put("longestMatchOnly", Boolean.FALSE);
130        params.put("wholeWordsOnly", Boolean.FALSE);
131    
132        if(gaz == null) {
133          gaz =
134            (LanguageAnalyser)Factory.createResource(
135              "gate.creole.gazetteer.DefaultGazetteer", params, hidden,
136              "Hashtag Tokenizer Gazetteer");
137        } else {
138          gaz.setParameterValues(params);
139          gaz.reInit();
140        }
141    
142        return this;
143      }

145      @Override
146      public void execute() throws ExecutionException {
147        AnnotationSet inputAS = document.getAnnotations(inputASName);
148        AnnotationSet outputAS = document.getAnnotations(outputASName);
149    
150        FeatureMap features = Factory.newFeatureMap();
151    
152        long startTime = System.currentTimeMillis();
153        fireStatusChanged("Tokenizing Hashtags: " + document.getName());
154        fireProgressChanged(0);
155        int count = 0;
156        
157        // get all the lookups we are going to use for decomposition...
158        AnnotationSet lookups = new AnnotationSetImpl(document);
159    
160        try {
161          // run the gazetteer to produce the HashtagLookup annotations
162          gaz.setParameterValue("annotationSetName", inputASName);
163          gaz.setDocument(document);
164          gaz.execute();
165    
166          // get all the hashtags
167          AnnotationSet hashtags = inputAS.get("Hashtag");
168    
169          for(Annotation hashtag : inputAS.get("Hashtag")) {
170            // for each hashtag in the document...
171    
172            AnnotationSet contained = inputAS.getContained(hashtag.getStartNode().getOffset(), hashtag.getEndNode().getOffset());
173            
174            //clear away any left overs from previous tags
175            lookups.clear();
176    
177            // which are the HashtagLookup
178            lookups.addAll(contained.get("HashtagLookup"));
179    
180            // any other Lookups the user has generated
181            lookups.addAll(contained.get("Lookup"));
182    
183            // and any number tokens
184            features = Factory.newFeatureMap();
185            features.put("kind", "number");
186            lookups.addAll(contained.get("Token", features));
187    
188            // the _ appears to be allowed so add them as well
189            features = Factory.newFeatureMap();
190            features.put("string", "_");
191            lookups.addAll(contained.get("Token", features));
192            
193            if(isInterrupted()) { throw new ExecutionInterruptedException(
194              "The execution of the hashtag tokenizer has been abruptly interrupted!"); }
195    
196            // this will hold the best we have seen so far
197            List<List<Annotation>> fewestTokens = new ArrayList<List<Annotation>>();
198    
199            // get all the lookups that start at the beginning of the hashtag
200            List<Annotation> start =
201              sort(getAnnotationsAtOffset(lookups, hashtag.getStartNode()
202                .getOffset() + 1));
203    
204            for(Annotation a : start) {
205              // for each lookup search for a valid tokenization
206              List<List<Annotation>> found =
207                search(lookups, hashtag.getEndNode().getOffset(), a);
208    
209              if(found != null) {
210                // if we found a contender and it's the best so far store it
211                if(fewestTokens.isEmpty()) {
212                  fewestTokens.addAll(found);
213                } else if(found.get(0).size() == fewestTokens.get(0).size()) {
214                  fewestTokens.addAll(found);
215                } else if(found.get(0).size() < fewestTokens.get(0).size()) {
216                  fewestTokens.clear();
217                  fewestTokens.addAll(found);
218                }
219              }
220            }
221    
222            if(debug && fewestTokens.size() > 1) {
223              System.out.println(stringFor(document, hashtag));
224              display(fewestTokens);
225            }
226    
227            if(fewestTokens.isEmpty()) {
228              // if we didn't find any sensible tokenizations then let's see if the
229              // hashtag is mized case
230    
231              String tagText = stringFor(document, hashtag).substring(1);
232              if("mixedCaps".equals(getTokenType(tagText)[1])) {
233                // if we have a mixed case hahstag then let's assume it is
234                // CamelCased and split it accordingly
235    
236                // TODO think about camel case which includes numbers
237    
238                // a list to hold the tokens
239                List<Annotation> found = new ArrayList<Annotation>();
240    
241                // start looking for token breaks aftert the initial #
242                long begin = hashtag.getStartNode().getOffset() + 1;
243    
244                for(String token : tagText
245                  .split("((?<=[a-z])(?=[A-Z]))|((?<=[A-Z]{2,})(?=[a-z]))")) {
246                  // split the token at the case changes...
247    
248                  // create the annotation in the Lookup set and add it to the found
249                  // list
250                  found.add(lookups.get(lookups.add(begin,
251                    (begin += token.length()), "CamelToken",
252                    Factory.newFeatureMap())));
253                }
254    
255                // record the tokenization so we can process it later
256                fewestTokens.add(found);
257              }
258    
259            }
260    
261            if(!fewestTokens.isEmpty()) {
262              // if we found a valid tokenization then...
263    
264              // remove any existing Token annotations
265              inputAS.removeAll(inputAS.get("Token").getContained(
266                hashtag.getStartNode().getOffset(),
267                hashtag.getEndNode().getOffset()));
268    
269              // create a punctuation Token over the #
270              features = Factory.newFeatureMap();
271              features.put("string", "#");
272              features.put("length", "1");
273              features.put("kind", "punctuation");
274              outputAS.add(hashtag.getStartNode().getOffset(), hashtag
275                .getStartNode().getOffset() + 1, "Token", features);
276    
277              // let's assume that the first one we found is best
278              int prefered = 0;
279    
280              for(int i = 0; i < fewestTokens.size(); ++i) {
281                // check those we have found and skip over any that contain single
282                // letter or mixed case words
283    
284                boolean okay = true;
285                for(Annotation a : fewestTokens.get(i)) {
286                  // single letter words are not great
287                  if(a.getEndNode().getOffset() - a.getStartNode().getOffset() == 1)
288                    okay = false;
289                }
290    
291                if(okay) {
292                  // if it contains neither a single letter word or a mixed case
293                  // word then we should definitely prefer this one
294                  prefered = i;
295                  break;
296                }
297              }
298    
299              for(Annotation a : fewestTokens.get(prefered)) {
300                // for each new token...
301    
302                // find where it starts/ends and its length
303                long startOffset = a.getStartNode().getOffset();
304                long endOffset = a.getEndNode().getOffset();
305                String length = Long.toString(endOffset - startOffset);
306    
307                // get the actual text
308                String string = stringFor(document, a);
309    
310                // work out what kind of token it is and if it is a word
311                // what its orthography is
312                String[] tokenType = getTokenType(string);
313                String kind = tokenType[0];
314                String orth = tokenType[1];
315    
316                // create the new Token annotation
317                features = Factory.newFeatureMap();
318                features.put("string", string);
319                features.put("length", length);
320                features.put("kind", kind);
321                if(orth != null) features.put("orth", orth);
322                outputAS.add(startOffset, endOffset, "Token", features);
323    
324                if(debug) {
325                  // for debug purposes add a matching set of HashtagToken
326                  // annotations
327                  features = Factory.newFeatureMap();
328                  features.put("string", string);
329                  features.put("length", length);
330                  features.put("kind", kind);
331                  if(orth != null) features.put("orth", orth);
332                  outputAS.add(startOffset, endOffset, "HashtagToken", features);
333                }
334              }
335            } else if(debug) {
336              System.err.println(stringFor(document, hashtag));
337              
338              AnnotationSet tokens = inputAS.get("Token").getContained(
339                hashtag.getStartNode().getOffset()+1,
340                hashtag.getEndNode().getOffset());
341              
342              for (Annotation token : tokens) {
343                
344                features = Factory.newFeatureMap();
345                features.putAll(token.getFeatures());
346                outputAS.add(token.getStartNode().getOffset(), token.getEndNode().getOffset(), "HashtagToken", features);
347              }
348            }
349    
350            fireProgressChanged(count++ * 100 / hashtags.size());
351          }
352    
353          fireProcessFinished();
354          fireStatusChanged("Hashtags in " +
355            document.getName() +
356            " tokenized in " +
357            NumberFormat.getInstance().format(
358              (double)(System.currentTimeMillis() - startTime) / 1000) +
359            " seconds!");
360        } catch(InvalidOffsetException e) {
361          throw new ExecutionException(e);
362        } catch(ResourceInstantiationException e) {
363          throw new ExecutionException(e);
364        } finally {
365          gaz.setDocument(null);
366    
367          if(!debug) {
368            inputAS.removeAll(inputAS.get("HashtagLookup"));
369          }
370        }
371      }

378      private static String[] getTokenType(String string) {
379        String kind = "symbol";
380        String orth = null;
381        if(isAlpha(string)) {
382          kind = "word";
383    
384          if(isAllLowerCase(string))
385            orth = "lowercase";
386          else if(isAllUpperCase(string))
387            orth = "allCaps";
388          else if(isAllUpperCase(string.substring(0, 1)) &&
389            isAllLowerCase(string.substring(1)))
390            orth = "upperInitial";
391          else orth = "mixedCaps";
392        } else if(isNumeric(string)) {
393          kind = "number";
394        } else if(string.matches("(\\p{Punct})+")) {
395          kind = "punctuation";
396        }
397    
398        return new String[]{kind, orth};
399      }

401      private void display(List<List<Annotation>> found) {
402        for(List<Annotation> tokens : found) {
403          System.out.print("   ");
404          for(Annotation token : tokens) {
405            System.out.print(stringFor(document, token) + " ");
406          }
407          System.out.print("\n");
408        }
409      }

411      /**
412       * Depth first search through a set of annotations to find a contiguous set
413       * for a given character range.
414       **/
415      private List<List<Annotation>> search(AnnotationSet lookups, Long endOffset,
416                                            Annotation token)
417        throws InvalidOffsetException {
418    
419        if("mixedCaps"
420          .equals(getTokenType(stringFor(lookups.getDocument(), token))[1]))
421          return null;
422    
423        List<List<Annotation>> shortest = new ArrayList<List<Annotation>>();
424    
425        if(token.getEndNode().getOffset().equals(endOffset)) {
426          // if the token we are starting from ends at the right place then we
427          // can stop and just return the single token as the result
428          List<Annotation> found = new ArrayList<Annotation>();
429          found.add(token);
430    
431          shortest.add(found);
432          return shortest;
433        }
434    
435        if(endOffset - token.getEndNode().getOffset() > 1) {
436          // if there are two or more characters after this token then...
437    
438          // get the rest of the text of the hashtag
439          String rest =
440            lookups.getDocument().getContent()
441              .getContent(token.getEndNode().getOffset() - 1, endOffset).toString();
442    
443          if(rest.substring(1).matches(rest.substring(0, 1) + "+")) {
444            // if the rest of the hashtag is just the same as the last letter of
445            // this token then someone has just lent on the keyboard for emphasis
446    
447            // so... extend the current token to include the rest of the hashtag
448            Annotation newToken =
449              lookups.get(lookups.add(token.getStartNode().getOffset(), endOffset,
450                "HashtagLookup", Factory.newFeatureMap()));
451    
452            // return this extended token
453            List<Annotation> found = new ArrayList<Annotation>();
454            found.add(newToken);
455    
456            shortest.add(found);
457            return shortest;
458          }
459        }
460    
461        // get all possibilities that start after the current annotation that
462        // could be used to cover the next bit of the document
463        List<Annotation> next =
464          sort(getAnnotationsAtOffset(lookups, token.getEndNode().getOffset()));
465    
466        // if there aren't any annotations we can use then return null as we can
467        // never fully cover the range from this point
468        if(next == null || next.isEmpty()) return null;
469    
470        for(Annotation a : next) {
471          // use each possible annotation to start a new search
472          List<List<Annotation>> part = search(lookups, endOffset, a);
473    
474          if(part != null) {
475            if(shortest.isEmpty()) {
476              shortest.addAll(part);
477            } else if(part.get(0).size() == shortest.get(0).size()) {
478              shortest.addAll(part);
479            } else if(part.get(0).size() < shortest.get(0).size()) {
480              shortest.clear();
481              shortest.addAll(part);
482            }
483          }
484        }
485    
486        // if we didn't find a match then return null
487        if(shortest.isEmpty()) return null;
488    
489        // add the token we started from to the beginning of the match
490        for(List<Annotation> found : shortest) {
491          found.add(0, token);
492        }
493    
494        // return the match we found
495        return shortest;
496      }

498      /**
499       * Returns a list in which the Annotations are sorted by length, longest
500       * first.
501       **/
502      private List<Annotation> sort(AnnotationSet annotations) {
503    
504        List<Annotation> sorted = new ArrayList<Annotation>();
505    
506        if(annotations == null || annotations.isEmpty()) return sorted;
507    
508        sorted.addAll(annotations);
509        Collections.sort(sorted, lengthComparator);
510    
511        // TODO filter out annotations with the same span
512    
513        List<Annotation> filtered = new ArrayList<Annotation>();
514        long length = -1;
515        for(Annotation a : sorted) {
516          long al = a.getEndNode().getOffset() - a.getStartNode().getOffset();
517          if(length == -1 || al != length) {
518            filtered.add(a);
519            length = al;
520          }
521        }
522    
523        return filtered;
524      }
